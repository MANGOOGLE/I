import RPi.GPIO as GPIO
import time, datetime
import telepot
from telepot.loop import MessageLoop

first_led, second_led, third_led, fourth_led = 7, 29, 31, 33
now = datetime.datetime.now()

GPIO.setmode(GPIO.BOARD)
GPIO.setwarnings(False)

for led in [first_led, second_led, third_led, fourth_led]:
    GPIO.setup(led, GPIO.OUT)
    GPIO.output(led, 0)

def action(msg):
    chat_id = msg['chat']['id']
    command = msg['text']
    print('Received:', command)

    if 'on' in command:
        message = "on "

        if '1' in command:
            GPIO.output(first_led,1); message += "first_led "
        if '2' in command:
            GPIO.output(second_led,1); message += "second_led "
        if '3' in command:
            GPIO.output(third_led,1); message += "third_led "
        if '4' in command:
            GPIO.output(fourth_led,1); message += "fourth_led "
        if 'all' in command:
            for led in [first_led,second_led,third_led,fourth_led]:
                GPIO.output(led,1)

        telegram_bot.sendMessage(chat_id, message+"light(s)")

    if 'off' in command:
        message = "off "

        if '1' in command:
            GPIO.output(first_led,0); message += "first_led "
        if '2' in command:
            GPIO.output(second_led,0); message += "second_led "
        if '3' in command:
            GPIO.output(third_led,0); message += "third_led "
        if '4' in command:
            GPIO.output(fourth_led,0); message += "fourth_led "
        if 'all' in command:
            for led in [first_led,second_led,third_led,fourth_led]:
                GPIO.output(led,0)

        telegram_bot.sendMessage(chat_id, message+"light(s)")

telegram_bot = telepot.Bot('fill in id.....')
print(telegram_bot.getMe())

MessageLoop(telegram_bot, action).run_as_thread()
print('up n runnin')

while True:
    time.sleep(10)
